#Область АсинхронныеОперации

Процедура НачатьФоновуюОперацию(ИмяМетода, ПараметрыМетода, ПараметрыОперации)
	
	ИдентификаторЗадания = FscFilestorageСервер.НачатьФоновуюОперацию(
		ИмяМетода,
		ПараметрыМетода,
		ПараметрыОперации.ПредставлениеОперации);
	
	ПараметрыФормы = Новый Структура;
	ПараметрыФормы.Вставить("ИдентификаторЗадания", ИдентификаторЗадания);
	ПараметрыФормы.Вставить("ВидОперации", ПараметрыОперации.ВидОперации);
	ПараметрыФормы.Вставить("ПредставлениеОперации", ПараметрыОперации.ПредставлениеОперации);
	
	Если ПараметрыОперации.Свойство("ПутьКФайлу") Тогда
		ПараметрыФормы.Вставить("ПутьКФайлу", ПараметрыОперации.ПутьКФайлу);
	КонецЕсли;
	
	Если ПараметрыОперации.Свойство("ПутьСохранения") Тогда
		ПараметрыФормы.Вставить("ПутьСохранения", ПараметрыОперации.ПутьСохранения);
	КонецЕсли;
	
	Контекст = Новый Структура;
	Контекст.Вставить("ОповещениеПриЗавершении", ПараметрыОперации.ОповещениеПриЗавершении);
	Контекст.Вставить("ВидОперации", ПараметрыОперации.ВидОперации);
	
	Если ПараметрыОперации.Свойство("ПутьСохранения") Тогда
		Контекст.Вставить("ПутьСохранения", ПараметрыОперации.ПутьСохранения);
	КонецЕсли;
	
	ОповещениеЗакрытия = Новый ОписаниеОповещения("НачатьФоновуюОперациюЗавершение", ЭтотОбъект, Контекст);
	
	ОткрытьФорму(
		"Обработка.FscFilestorageОбработка.Форма",
		ПараметрыФормы,
		,
		,
		,
		,
		ОповещениеЗакрытия,
		РежимОткрытияОкнаФормы.БлокироватьОкноВладельца);
	
КонецПроцедуры

Процедура НачатьФоновуюОперациюЗавершение(Результат, Параметры) Экспорт
	
	РезультатОповещения = РезультатДляОповещения(Результат, Параметры);
	
	Если Параметры.ОповещениеПриЗавершении <> Неопределено Тогда
		ВыполнитьОбработкуОповещения(Параметры.ОповещениеПриЗавершении, РезультатОповещения);
	КонецЕсли;
	
КонецПроцедуры

Функция РезультатДляОповещения(РезультатОперации, Параметры)
	
	Если Параметры.ВидОперации = "Download" Тогда
		Успешно = ЗначениеРеквизита(РезультатОперации, "Успешно");
		Если Успешно = Истина Тогда
			Возврат Параметры.ПутьСохранения;
		КонецЕсли;
		
		Возврат РезультатОперации;
	КонецЕсли;
	
	Если Параметры.ВидОперации = "Upload" Тогда
		ИдентификаторФайла = ИдентификаторЗагруженногоФайла(РезультатОперации);
		Если ЗначениеЗаполнено(ИдентификаторФайла) Тогда
			Возврат ИдентификаторФайла;
		КонецЕсли;
	КонецЕсли;
	
	Возврат РезультатОперации;
	
КонецФункции

Функция ИдентификаторЗагруженногоФайла(РезультатОперации)
	
	Если РезультатОперации = Неопределено Тогда
		Возврат Неопределено;
	КонецЕсли;
	
	Данные = ЗначениеРеквизита(РезультатОперации, "Данные");
	Если Данные <> Неопределено Тогда
		
		Идентификатор = ЗначениеРеквизита(Данные, "id");
		Если ЗначениеЗаполнено(Идентификатор) Тогда
			Возврат Идентификатор;
		КонецЕсли;
		
		Идентификатор = ЗначениеРеквизита(Данные, "fileId");
		Если ЗначениеЗаполнено(Идентификатор) Тогда
			Возврат Идентификатор;
		КонецЕсли;
		
		Идентификатор = ЗначениеРеквизита(Данные, "fileGuid");
		Если ЗначениеЗаполнено(Идентификатор) Тогда
			Возврат Идентификатор;
		КонецЕсли;
		
		Идентификатор = ЗначениеРеквизита(Данные, "guid");
		Если ЗначениеЗаполнено(Идентификатор) Тогда
			Возврат Идентификатор;
		КонецЕсли;
	КонецЕсли;
	
	Возврат Неопределено;
	
КонецФункции

Функция ЗначениеРеквизита(Данные, ИмяРеквизита)
	
	НайденноеЗначение = Неопределено;
	
	Попытка
		Если Данные.Свойство(ИмяРеквизита, НайденноеЗначение) Тогда
			Возврат НайденноеЗначение;
		КонецЕсли;
	Исключение
	КонецПопытки;
	
	Попытка
		Возврат Данные[ИмяРеквизита];
	Исключение
	КонецПопытки;
	
	Возврат Неопределено;
