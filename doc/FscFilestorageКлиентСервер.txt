#Область JSON

Функция ВJSON(Данные) Экспорт
	Запись = Новый ЗаписьJSON;
	Запись.УстановитьСтроку();
	ЗаписатьJSON(Запись, Данные);
	Возврат Запись.Закрыть();
КонецФункции

Функция ИзJSON(ТекстJSON) Экспорт
	Если ПустаяСтрока(ТекстJSON) Тогда
		Возврат Неопределено;
	КонецЕсли;

	Чтение = Новый ЧтениеJSON;
	Чтение.УстановитьСтроку(ТекстJSON);
	Возврат ПрочитатьJSON(Чтение);
КонецФункции

Функция РезультатHTTP(КодСостояния, Тело, Заголовки = Неопределено) Экспорт
	Результат = Новый Структура;
	Результат.Вставить("КодСостояния", КодСостояния);
	Результат.Вставить("Успешно", КодСостояния >= 200 И КодСостояния < 300);
	Результат.Вставить("Тело", Тело);
	Результат.Вставить("Заголовки", Заголовки);

	Если ЗначениеЗаполнено(Тело) Тогда
		Попытка
			Результат.Вставить("Данные", ИзJSON(Тело));
		Исключение
			Результат.Вставить("Данные", Неопределено);
		КонецПопытки;
	Иначе
		Результат.Вставить("Данные", Неопределено);
	КонецЕсли;

	Возврат Результат;
КонецФункции

Функция MIMEТипПоИмениФайла(ИмяФайла) Экспорт
	Расширение = НРег(РасширениеФайла(ИмяФайла));

	Если Расширение = ".txt" Тогда
		Возврат "text/plain";
	ИначеЕсли Расширение = ".json" Тогда
		Возврат "application/json";
	ИначеЕсли Расширение = ".pdf" Тогда
		Возврат "application/pdf";
	ИначеЕсли Расширение = ".jpg" Или Расширение = ".jpeg" Тогда
		Возврат "image/jpeg";
	ИначеЕсли Расширение = ".png" Тогда
		Возврат "image/png";
	ИначеЕсли Расширение = ".zip" Тогда
		Возврат "application/zip";
	КонецЕсли;

	Возврат "application/octet-stream";
КонецФункции

Функция РасширениеФайла(ИмяФайла) Экспорт
	Позиция = СтрДлина(ИмяФайла);
	Пока Позиция > 0 Цикл
		Символ = Сред(ИмяФайла, Позиция, 1);
		Если Символ = "." Тогда
			Возврат Сред(ИмяФайла, Позиция);
		КонецЕсли;
		Если Символ = "/" Или Символ = "\" Тогда
			Прервать;
		КонецЕсли;
		Позиция = Позиция - 1;
	КонецЦикла;

	Возврат "";
КонецФункции
