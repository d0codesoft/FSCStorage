#Область Соединение

Функция HTTPСоединение() Экспорт
	Адрес = FscFilestorageКлиентСервер.АдресСервиса();
	ЧастиURL = РазобратьАдресСервиса(Адрес);

	ЗащищенноеСоединение = ЗащищенноеСоединениеБезПроверкиСертификата();

	Возврат Новый HTTPСоединение(
		ЧастиURL.Сервер,
		ЧастиURL.Порт,
		,
		,
		,
		FscFilestorageКлиентСервер.ТаймаутHTTP(),
		ЗащищенноеСоединение);
КонецФункции

Функция РазобратьАдресСервиса(Адрес)
	РабочийАдрес = СтрЗаменить(Адрес, "https://", "");
	РабочийАдрес = СтрЗаменить(РабочийАдрес, "http://", "");

	ПозицияСлеша = Найти(РабочийАдрес, "/");
	Если ПозицияСлеша > 0 Тогда
		РабочийАдрес = Лев(РабочийАдрес, ПозицияСлеша - 1);
	КонецЕсли;

	Порт = 443;
	Сервер = РабочийАдрес;

	ПозицияПорта = Найти(РабочийАдрес, ":");
	Если ПозицияПорта > 0 Тогда
		Сервер = Лев(РабочийАдрес, ПозицияПорта - 1);
		Порт = Число(Сред(РабочийАдрес, ПозицияПорта + 1));
	КонецЕсли;

	Возврат Новый Структура("Сервер,Порт", Сервер, Порт);
КонецФункции

Функция ЗащищенноеСоединениеБезПроверкиСертификата()
	// Для разных релизов платформы 8.3 сигнатуры конструктора отличаются.
	// Смысл один: HTTPS через OpenSSL без проверки цепочки сертификата.
	Попытка
		Возврат Новый ЗащищенноеСоединениеOpenSSL(Неопределено, Неопределено, Неопределено, Ложь);
	Исключение
		Попытка
			Возврат Новый ЗащищенноеСоединениеOpenSSL(Неопределено, Неопределено);
		Исключение
			Возврат Новый ЗащищенноеСоединениеOpenSSL();
		КонецПопытки;
	КонецПопытки;
КонецФункции

Процедура ЗаполнитьЗаголовкиАвторизации(Запрос, ApiToken, TenantGuid = Неопределено)
	Запрос.Заголовки.Вставить("X-Api-Token", ApiToken);

	Если ЗначениеЗаполнено(TenantGuid) Тогда
		Запрос.Заголовки.Вставить("X-Tenant-Id", Строка(TenantGuid));
	КонецЕсли;

Функция ВыполнитьJSON(Метод, URL, Данные, ApiToken, TenantGuid = Неопределено) Экспорт
	Соединение = HTTPСоединение();
	Запрос = Новый HTTPЗапрос(URL);
	ЗаполнитьЗаголовкиАвторизации(Запрос, ApiToken, TenantGuid);
	Запрос.Заголовки.Вставить("Accept", "application/json");
	Запрос.Заголовки.Вставить("Content-Type", "application/json; charset=utf-8");

	Если Данные <> Неопределено Тогда
		Запрос.УстановитьТелоИзСтроки(FscFilestorageКлиентСервер.ВJSON(Данные), КодировкаТекста.UTF8);
	КонецЕсли;

	Ответ = Соединение.ВызватьHTTPМетод(Метод, Запрос);
	Возврат FscFilestorageКлиентСервер.РезультатHTTP(
		Ответ.КодСостояния,
		Ответ.ПолучитьТелоКакСтроку(),
		Ответ.Заголовки);

Функция ВыполнитьMultipart(Метод, URL, ЧастиФормы, ApiToken, TenantGuid = Неопределено) Экспорт
	Boundary = FscFilestorageКлиентСервер.НовыйBoundary();
	ИмяВременногоФайла = ПолучитьИмяВременногоФайла("tmp");

	Запись = Новый ЗаписьДанных(ИмяВременногоФайла);

	Для Каждого Часть Из ЧастиФормы Цикл
		Запись.ЗаписатьСтроку("--" + Boundary + FscFilestorageКлиентСервер.CRLF());

		Если Часть.Свойство("ПутьКФайлу") Тогда
			ИмяФайла = ?(Часть.Свойство("ИмяФайла"), Часть.ИмяФайла, FscFilestorageКлиентСервер.ИмяФайлаИзПути(Часть.ПутьКФайлу));
			ContentType = ?(Часть.Свойство("ContentType"), Часть.ContentType, FscFilestorageКлиентСервер.MIMEТипПоИмениФайла(ИмяФайла));

			Запись.ЗаписатьСтроку(
				"Content-Disposition: form-data; name=""" + Часть.Имя + """; filename=""" + ИмяФайла + """" + FscFilestorageКлиентСервер.CRLF());
			Запись.ЗаписатьСтроку("Content-Type: " + ContentType + FscFilestorageКлиентСервер.CRLF());
			Запись.ЗаписатьСтроку(FscFilestorageКлиентСервер.CRLF());
			Запись.Записать(Новый ДвоичныеДанные(Часть.ПутьКФайлу));
			Запись.ЗаписатьСтроку(FscFilestorageКлиентСервер.CRLF());
		Иначе
			Запись.ЗаписатьСтроку(
				"Content-Disposition: form-data; name=""" + Часть.Имя + """" + FscFilestorageКлиентСервер.CRLF());
			Запись.ЗаписатьСтроку(FscFilestorageКлиентСервер.CRLF());
			Запись.ЗаписатьСтроку(Строка(Часть.Значение) + FscFilestorageКлиентСервер.CRLF());
		КонецЕсли;
	КонецЦикла;

	Запись.ЗаписатьСтроку("--" + Boundary + "--" + FscFilestorageКлиентСервер.CRLF());
	Запись.Закрыть();

	Результат = Неопределено;

	Попытка
		Соединение = HTTPСоединение();
		Запрос = Новый HTTPЗапрос(URL);
		ЗаполнитьЗаголовкиАвторизации(Запрос, ApiToken, TenantGuid);
		Запрос.Заголовки.Вставить("Accept", "application/json");
		Запрос.Заголовки.Вставить("Content-Type", "multipart/form-data; boundary=" + Boundary);
		Запрос.УстановитьИмяФайлаТела(ИмяВременногоФайла);

		Ответ = Соединение.ВызватьHTTPМетод(Метод, Запрос);
		Результат = FscFilestorageКлиентСервер.РезультатHTTP(
			Ответ.КодСостояния,
			Ответ.ПолучитьТелоКакСтроку(),
			Ответ.Заголовки);
	Исключение
		Попытка
			УдалитьФайлы(ИмяВременногоФайла);
		Исключение
		КонецПопытки;
		ВызватьИсключение;
	КонецПопытки;

	Попытка
		УдалитьФайлы(ИмяВременногоФайла);
	Исключение
	КонецПопытки;

	Возврат Результат;
КонецФункции

Функция ЗагрузитьФайл(ПутьКФайлу, ApiToken, TenantGuid, Категория = Неопределено, ВнешнийКлюч = Неопределено) Экспорт
	Части = Новый Массив;

	Если ЗначениеЗаполнено(ВнешнийКлюч) Тогда
		Части.Добавить(Новый Структура("Имя,Значение", "externalKey", ВнешнийКлюч));

Функция СкачатьФайл(ФайлGuid, ПутьСохранения, ApiToken, TenantGuid) Экспорт
	Соединение = HTTPСоединение();
	Запрос = Новый HTTPЗапрос(FscFilestorageКлиентСервер.URLСкачиванияФайла(ФайлGuid));
	ЗаполнитьЗаголовкиАвторизации(Запрос, ApiToken, TenantGuid);

	Ответ = Соединение.Получить(Запрос);

	Если Ответ.КодСостояния >= 200 И Ответ.КодСостояния < 300 Тогда
		Ответ.ПолучитьТелоКакДвоичныеДанные().Записать(ПутьСохранения);
	КонецЕсли;

	Возврат FscFilestorageКлиентСервер.РезультатHTTP(
		Ответ.КодСостояния,
		?(Ответ.КодСостояния >= 200 И Ответ.КодСостояния < 300, "", Ответ.ПолучитьТелоКакСтроку()),
		Ответ.Заголовки);
КонецФункции

#Область MultipartAPI

Функция ИнициализироватьMultipart(ПутьКФайлу, TenantId, ApiToken, TenantGuid, РазмерЧасти = Неопределено, ОжидаемыйSha256 = Неопределено) Экспорт
	Если РазмерЧасти = Неопределено Тогда
		РазмерЧасти = FscFilestorageКлиентСервер.РазмерЧастиПоУмолчанию();
	КонецЕсли;

	ИмяФайла = FscFilestorageКлиентСервер.ИмяФайлаИзПути(ПутьКФайлу);

	Тело = Новый Структура;
	Тело.Вставить("fileName", ИмяФайла);
	Тело.Вставить("fileSize", FscFilestorageКлиентСервер.РазмерФайла(ПутьКФайлу));
	Тело.Вставить("contentType", FscFilestorageКлиентСервер.MIMEТипПоИмениФайла(ИмяФайла));
	Тело.Вставить("partSize", РазмерЧасти);
	Тело.Вставить("tenantId", Строка(TenantId));

	Если ЗначениеЗаполнено(ОжидаемыйSha256) Тогда
		Тело.Вставить("expectedChecksumSha256", ОжидаемыйSha256);
	КонецЕсли;

	Возврат ВыполнитьJSON("POST", FscFilestorageКлиентСервер.URLMultipartInit(), Тело, ApiToken, TenantGuid);
КонецФункции

Функция ЗагрузитьЧастьMultipart(UploadId, НомерЧасти, ПутьКФайлуЧасти, ApiToken, TenantGuid, Sha256Части = Неопределено) Экспорт

		"Имя,ПутьКФайлу,ИмяФайла,ContentType",

		ПутьКФайлуЧасти,
		FscFilestorageКлиентСервер.ИмяФайлаИзПути(ПутьКФайлуЧасти),
		"application/octet-stream"));

		"PUT",
		FscFilestorageКлиентСервер.URLMultipartPart(UploadId, НомерЧасти),
