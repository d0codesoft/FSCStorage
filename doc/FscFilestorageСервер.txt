Процедура УстановкаПараметровСеанса(ТребуемыеПараметры) Экспорт
	Если ТребуемыеПараметры = Неопределено Тогда
		// раздел "безусловной" инициализации параметров сеанса
		ПараметрыСеанса.FilestorageState = Константы.FscFilestorageState.Получить();
		ПараметрыСеанса.FilestorageHost = Константы.FscFilestorageHost.Получить();
		ПараметрыСеанса.FilestorageApiToken = Константы.FscFilestorageApiToken.Получить();
		ПараметрыСеанса.FilestorageTenantId = Константы.FscFilestorageTenantId.Получить();
		
	Иначе		
		
		Для Каждого ИмяПараметра ИЗ ТребуемыеПараметры Цикл
			
			Если  ИмяПараметра = "FilestorageState" Тогда
				ПараметрыСеанса.FilestorageState = Константы.FscFilestorageState.Получить();
			ИначеЕсли  ИмяПараметра = "FilestorageHost" Тогда
				ПараметрыСеанса.FilestorageHost = Константы.FscFilestorageHost.Получить();
			ИначеЕсли  ИмяПараметра = "FilestorageApiToken" Тогда
				ПараметрыСеанса.FilestorageApiToken = Константы.FscFilestorageApiToken.Получить();
			ИначеЕсли  ИмяПараметра = "FilestorageTenantId" Тогда
				ПараметрыСеанса.FilestorageTenantId = Константы.FscFilestorageTenantId.Получить();
			КонецЕсли;		
			
		КонецЦикла;
		
	КонецЕсли;

Функция HTTPСоединение(ПараметрыСоединения) Экспорт
	Адрес = ПараметрыСоединения.Host;
	ЧастиURL = РазобратьАдресСервиса(Адрес);

	ЗащищенноеСоединение = Неопределено;
	Если ЧастиURL.Схема = "https" Тогда
		ЗащищенноеСоединение = ЗащищенноеСоединениеБезПроверкиСертификата();
	КонецЕсли;

	Возврат Новый HTTPСоединение(
		ЧастиURL.Сервер,
		ЧастиURL.Порт,
		,
		,
		,
		ТаймаутHTTP(),
		ЗащищенноеСоединение);

Функция РазобратьАдресСервиса(Адрес)
	Схема = "https";
	РабочийАдрес = Адрес;

	Если Лев(НРег(РабочийАдрес), 8) = "https://" Тогда
		Схема = "https";
		РабочийАдрес = Сред(РабочийАдрес, 9);
	ИначеЕсли Лев(НРег(РабочийАдрес), 7) = "http://" Тогда
		Схема = "http";
		РабочийАдрес = Сред(РабочийАдрес, 8);
	КонецЕсли;

	ПозицияСлеша = Найти(РабочийАдрес, "/");
	Если ПозицияСлеша > 0 Тогда
		РабочийАдрес = Лев(РабочийАдрес, ПозицияСлеша - 1);
	КонецЕсли;

	Порт = ?(Схема = "https", 443, 80);
	Сервер = РабочийАдрес;

	ПозицияПорта = Найти(РабочийАдрес, ":");
	Если ПозицияПорта > 0 Тогда
		Сервер = Лев(РабочийАдрес, ПозицияПорта - 1);
		Порт = Число(Сред(РабочийАдрес, ПозицияПорта + 1));
	КонецЕсли;

	Возврат Новый Структура("Схема,Сервер,Порт", Схема, Сервер, Порт);
КонецФункции

	Попытка
		Возврат Новый ЗащищенноеСоединениеOpenSSL(Неопределено, Неопределено);
	Исключение

		Исключение
			Возврат Новый ЗащищенноеСоединениеOpenSSL();
		КонецПопытки;
	КонецПопытки;

Функция ВыполнитьMultipart(Метод, URL, ЧастиФормы) Экспорт
	Boundary = FscFilestorageКлиентСервер.НовыйBoundary();
	ИмяВременногоФайла = ПолучитьИмяВременногоФайла("tmp");

	Запись = Новый ЗаписьДанных(ИмяВременногоФайла);

	Для Каждого Часть Из ЧастиФормы Цикл
		Запись.ЗаписатьСтроку("--" + Boundary + FscFilestorageКлиентСервер.CRLF());

		Если Часть.Свойство("ПутьКФайлу") Тогда
			ИмяФайла = ?(Часть.Свойство("ИмяФайла"), Часть.ИмяФайла, FscFilestorageКлиентСервер.ИмяФайлаИзПути(Часть.ПутьКФайлу));
			ContentType = ?(Часть.Свойство("ContentType"), Часть.ContentType, FscFilestorageКлиентСервер.MIMEТипПоИмениФайла(ИмяФайла));

			Запись.ЗаписатьСтроку(
				"Content-Disposition: form-data; name=""" + Часть.Имя + """; filename=""" + ИмяФайла + """" + FscFilestorageКлиентСервер.CRLF());
			Запись.ЗаписатьСтроку("Content-Type: " + ContentType + FscFilestorageКлиентСервер.CRLF());
			Запись.ЗаписатьСтроку(FscFilestorageКлиентСервер.CRLF());
			Запись.Записать(Новый ДвоичныеДанные(Часть.ПутьКФайлу));
			Запись.ЗаписатьСтроку(FscFilestorageКлиентСервер.CRLF());
		Иначе
			Запись.ЗаписатьСтроку(
				"Content-Disposition: form-data; name=""" + Часть.Имя + """" + FscFilestorageКлиентСервер.CRLF());
			Запись.ЗаписатьСтроку(FscFilestorageКлиентСервер.CRLF());
			Запись.ЗаписатьСтроку(Строка(Часть.Значение) + FscFilestorageКлиентСервер.CRLF());
		КонецЕсли;
	КонецЦикла;

	Запись.ЗаписатьСтроку("--" + Boundary + "--" + FscFilestorageКлиентСервер.CRLF());
	Запись.Закрыть();

	Результат = Неопределено;

	Попытка             
		Параметры = ПолучитьПараметрыСоединения();
		Соединение = HTTPСоединение(Параметры);
		Запрос = Новый HTTPЗапрос(URL);
		ЗаполнитьЗаголовкиАвторизации(Запрос, Параметры.ApiToken, Параметры.TenantId);
		Запрос.Заголовки.Вставить("Accept", "application/json");
		Запрос.Заголовки.Вставить("Content-Type", "multipart/form-data; boundary=" + Boundary);
		Запрос.УстановитьИмяФайлаТела(ИмяВременногоФайла);

		Ответ = Соединение.ВызватьHTTPМетод(Метод, Запрос);
		Результат = FscFilestorageКлиентСервер.РезультатHTTP(
			Ответ.КодСостояния,
			Ответ.ПолучитьТелоКакСтроку(),
			Ответ.Заголовки);
	Исключение                            
		
		ЗаписьЖурналаРегистрации("FileStorage",УровеньЖурналаРегистрации.Ошибка, Неопределено, ИмяВременногоФайла, "Помилка виконання запиту Multipart:"+ОписаниеОшибки());
		Попытка
			УдалитьФайлы(ИмяВременногоФайла);
		Исключение
		КонецПопытки;
		ВызватьИсключение;
	КонецПопытки;

	Попытка
		УдалитьФайлы(ИмяВременногоФайла);
	Исключение
	КонецПопытки;

	Возврат Результат;

Функция ИнициализироватьMultipart(ПутьКФайлу, РазмерЧасти = Неопределено) Экспорт
	Если РазмерЧасти = Неопределено Тогда
		РазмерЧасти = FscFilestorageКлиентСервер.РазмерЧастиПоУмолчанию();
	КонецЕсли;
	
	Параметры = ПолучитьПараметрыСоединения();
	
	ИмяФайла = FscFilestorageКлиентСервер.ИмяФайлаИзПути(ПутьКФайлу);

	Тело = Новый Структура;
	Тело.Вставить("fileName", ИмяФайла);
	Тело.Вставить("fileSize", FscFilestorageКлиентСервер.РазмерФайла(ПутьКФайлу));
	Тело.Вставить("contentType", FscFilestorageКлиентСервер.MIMEТипПоИмениФайла(ИмяФайла));
	Тело.Вставить("partSize", РазмерЧасти);
	Тело.Вставить("tenantId", Строка(Параметры.TenantId));
	
	Хешер = Новый ХешированиеДанных(ХешФункция.SHA256);
	ДвоичныеДанныеФайла = Новый ДвоичныеДанные(ПутьКФайлу);
	Хешер.Добавить(ДвоичныеДанныеФайла);

	ХешСумма = Хешер.ХешСумма;
	ХешСтрокой = Строка(ХешСумма);
	ХешСтрокой = СтрЗаменить(ХешСтрокой, " ", "");
	
	ДвоичныеДанныеФайла = Неопределено;
	Хешер = Неопределено;

	Тело.Вставить("expectedChecksumSha256", ХешСтрокой);

	Возврат ВыполнитьJSON("POST", FscFilestorageКлиентСервер.URLMultipartInit(), Тело);
