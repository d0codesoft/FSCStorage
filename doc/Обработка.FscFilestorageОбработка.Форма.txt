
&НаКлиенте
Процедура ПроверкаФоновогоЗадания()
	
	Результат=ПроверкаФоновогоЗаданияНаСервере();
	Если Результат = Неопределено Тогда
		ЗавершитьФоновуюОперацию(Неопределено);
	Иначе
		Если НЕ Результат.Состояние = "Задание выполняется" Тогда
			ЗавершитьФоновуюОперацию(Результат);
		КонецЕсли;
	КонецЕсли;
КонецПроцедуры

&НаСервере
Функция ПроверкаФоновогоЗаданияНаСервере()
	Результат = Новый Структура("Прогресс,Состояние");
		
	Задание = ФоновыеЗадания.НайтиПоУникальномуИдентификатору(ИдентификаторЗадания);
	Если Задание = Неопределено Тогда
		Возврат Неопределено;
	Иначе
		ЗаполнитьЗначенияСвойств(Результат,Задание);
		Результат.Состояние= Строка(Результат.Состояние);
		
		Сообщения=Задание.ПолучитьСообщенияПользователю(Ложь);
		Для Каждого Сообщение ИЗ Сообщения Цикл
			Если Лев(Сообщение.Текст,1) <> "{" Тогда
				Сообщить(Сообщение.Текст);
			КонецЕсли;
		КонецЦикла;
		Прогресс = ДлительныеОперации.ПрочитатьПрогресс(ИдентификаторЗадания);
		Если Прогресс <> Неопределено Тогда
			ПрогрессОбработки = Прогресс.Процент;//?(Прогресс.Процент<>0,Прогресс.Процент,ПрогрессОбработки);
		КонецЕсли;
		Результат.Вставить("Прогресс",Прогресс);
		
		Если НЕ Результат.Состояние = "Задание выполняется" Тогда
			Попытка
				Результат.Вставить("РезультатОперации", Задание.ПолучитьРезультат());
			Исключение
				Результат.Вставить("РезультатОперации", Неопределено);
				Результат.Вставить("ТекстОшибки", ОписаниеОшибки());
			КонецПопытки;
		КонецЕсли;
	КонецЕсли;

	Возврат Результат;
КонецФункции

&НаКлиенте
Процедура УбитьПроцесс(Команда)
	
	УбитьПроцессНаСервере();
	ОтключитьОбработчикОжидания("ПроверкаФоновогоЗадания");
	Сообщить(""+ТекущаяДата()+" задание отменено");
	Закрыть(Неопределено);
	
КонецПроцедуры

&НаСервере
Процедура УбитьПроцессНаСервере()
	Задание = ФоновыеЗадания.НайтиПоУникальномуИдентификатору(ИдентификаторЗадания);
	Если Задание <> Неопределено Тогда
		Задание.Отменить();
	КонецЕсли;
	Элементы.ГруппаФоновоеЗадание.Видимость=Ложь;
КонецПроцедуры    

&НаСервере
Процедура ПриСозданииНаСервере(Отказ, СтандартнаяОбработка)
	
	Если Параметры.Свойство("ИдентификаторЗадания") Тогда
		ИдентификаторЗадания = Параметры.ИдентификаторЗадания;
	КонецЕсли;
	
	Элементы.ГруппаФоновоеЗадание.Видимость=Истина;
	
КонецПроцедуры

&НаКлиенте
Процедура ПриОткрытии(Отказ)
	ПодключитьОбработчикОжидания("ПроверкаФоновогоЗадания", 1);
КонецПроцедуры

&НаКлиенте
Процедура ЗавершитьФоновуюОперацию(РезультатЗадания)
	
	ОтключитьОбработчикОжидания("ПроверкаФоновогоЗадания");
	
	РезультатОперации = Неопределено;
	Если РезультатЗадания <> Неопределено И РезультатЗадания.Свойство("РезультатОперации") Тогда
		РезультатОперации = РезультатЗадания.РезультатОперации;
	КонецЕсли;
	
	Сообщить(""+ТекущаяДата()+" задание выполнено");
	Закрыть(РезультатОперации);
	
КонецПроцедуры
